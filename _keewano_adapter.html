Why this page?

Many games already use a common IAnalytics interface that forwards the same context (device, session, A/B group, campaign, etc.) on every analytics call to tools like Mixpanel, GameAnalytics, etc.

Keewano is context-first: once the SDK knows user identity, language, A/B assignment, campaign, etc., you don't repeat them per event - the SDK reuses the context on the user's timeline.
This page shows a drop-in adapter so you can:

  • Keep your existing interface and call sites unchanged
  • Continue sending "rich" context to your interface
  • Let the Keewano adapter extract only what Keewano needs, and only when it changes

A "rich" interface (typical of legacy tools)

Your interface may insist on passing context for every call.
Keep it as-is. We will implement a Keewano-backed version that accepts all fields but forwards only the relevant ones.

}

Keewano adapter: accept everything, forward only what's needed

The adapter caches the last context and only calls the Keewano setters when values change.
Noise fields (device, session, platform, OS, screen) are ignored because Keewano auto-captures them.

public sealed class KeewanoAnalytics : IAnalytics
{
private AnalyticsContext lastCtx;
// -------- Identity & consent --------
public void SetConsent(AnalyticsContext ctx, bool granted)
{ KeewanoSDK.SetUserConsent(granted); }
// -------- UI & UX --------
public void WindowOpen(AnalyticsContext ctx, string windowName)
{ KeewanoSDK.ReportWindowOpen(windowName); }
public void WindowClose(AnalyticsContext ctx, string windowName)
{ KeewanoSDK.ReportWindowClose(windowName); }
public void ButtonClick(AnalyticsContext ctx, string buttonId)
{ KeewanoSDK.ReportButtonClick(buttonId); }
// -------- FTUE --------
public void OnboardingStep(AnalyticsContext ctx, string stepName)
{ KeewanoSDK.ReportOnboardingMilestone(stepName); }
// -------- Gameplay --------
public void LevelStart(AnalyticsContext ctx, uint level, string difficulty = null)
{ KeewanoSDK.ReportLevelStart(level); }
public void LevelWin(AnalyticsContext ctx, uint level)
{ KeewanoSDK.ReportLevelWin(level); }
public void LevelLose(AnalyticsContext ctx, uint level)
{ KeewanoSDK.ReportLevelLose(level); }
// -------- Economy --------
public void ItemsReset(AnalyticsContext ctx, string reason, IEnumerable<ItemChange> items)
{ KeewanoSDK.ReportItemsReset(reason, Map(items)); }
public void ItemsExchange(AnalyticsContext ctx, string reason,
IEnumerable<ItemChange> from, IEnumerable<ItemChange> to)
{ KeewanoSDK.ReportItemsExchange(reason, Map(from), Map(to)); }
private static Keewano.Item[] Map(IEnumerable<ItemChange> src)
=> src == null ? null : src.Select(i => new Keewano.Item(i.Name, i.Quantity)).ToArray();
// -------- Monetization --------
public void IapSuccess(AnalyticsContext ctx, string productId, uint priceCents, string currency = null)
{ KeewanoSDK.ReportInAppPurchase(productId, priceCents); }
}
